    async function postReq(payload) {

        return fetch(API_URL, {

            method: 'POST',

            redirect: 'follow',

            headers: { 'Content-Type': 'text/plain;charset=utf-8' },

    async function fetchAll() {

        try {

            const res = await fetch(API_URL);

            const data = await res.json();


            // ── 수정 1: App Script가 서버사이드에서 가져온 지표 데이터 사용 ──

            if (data.indices) {

                Object.entries(data.indices).forEach(([sym, info]) => {

                    cache[sym] = {

                        price:  info.price,

                        change: info.change,

                        name:   mSym[sym] || info.name,

                        curr:   info.currency === 'KRW' ? 'KRW' : 'USD'

                    };

                });

            }


            if (data.trades) {

                users.forEach(u => state.trades[u] = data.trades.filter(t => t.user === u));

            }

        } catch(e) {

            console.error("스프레드시트 데이터 연동 실패:", e);

        }


        // 환율 업데이트

        if (cache['KRW=X'] && cache['KRW=X'].price > 0) exKRW = cache['KRW=X'].price;


        // 종목 현재가 캐시 보충 (App Script가 안 가져온 종목만)

        let missingTickers = new Set();

        users.forEach(u => state.trades[u].forEach(t => {

            if (!cache[t.ticker] || cache[t.ticker].price <= 0) missingTickers.add(t.ticker);

        }));


        if (missingTickers.size > 0) {

            const q = Array.from(missingTickers).map(t => /^\d{6}$/.test(t) ? t+'.KS' : t).join(',');

            try {

                const url = encodeURIComponent(`https://query1.finance.yahoo.com/v7/finance/quote?symbols=${q}`);

                const yRes = await fetch(`https://api.allorigins.win/get?url=${url}`);

                const yData = await yRes.json();

                const parsed = JSON.parse(yData.contents);

                if (parsed.quoteResponse && parsed.quoteResponse.result) {

                    parsed.quoteResponse.result.forEach(item => {

                        const tk = item.symbol.replace('.KS', '');

                        cache[tk] = {

                            price:  item.regularMarketPrice || 0,

                            change: item.regularMarketChangePercent || 0,

                            name:   item.shortName || item.longName || tMap[tk] || tk,

                            curr:   item.currency === 'KRW' ? 'KRW' : 'USD'

                        };

                    });

                }

            } catch(e) {

                console.warn("보조 야후 API 조회 실패:", e);

            }

        }


        renderIdx();

        updateAll();

    }


    function renderIdx() {

        const grid = document.getElementById('market-indices');

        const html = Object.entries(mSym).map(([sym, lbl]) => {

            const info = cache[sym];

            let priceStr = '-';

            let changeHtml = '<span class="pct" style="color:#888">(-)</span>';


            if (info && info.price > 0) {

                priceStr = info.price.toLocaleString(undefined, { maximumFractionDigits: 2 });

                const pct = info.change;

                const cls  = pct > 0 ? 'profit' : (pct < 0 ? 'loss' : '');

                const sign = pct > 0 ? '+' : '';

                changeHtml = `<span class="pct ${cls}">(${sign}${pct.toFixed(2)}%)</span>`;

            }


            return `<div class="idx-item">

                        <div class="val">${priceStr} ${changeHtml}</div>

                        <div class="key">${lbl}</div>

                    </div>`;

        }).join('');


        grid.innerHTML = html;

    }


    async function getStk(tk) {

        if (!tk) return null;

        if (cache[tk] && cache[tk].price > 0) return cache[tk];


        let qTk = /^\d{6}$/.test(tk) ? tk+'.KS' : tk;

        try {

            const url = encodeURIComponent(`https://query1.finance.yahoo.com/v8/finance/chart/${qTk}`);

            const res  = await fetch(`https://api.allorigins.win/get?url=${url}`);

            const data = await res.json();

            const meta = JSON.parse(data.contents).chart.result[0].meta;


            const price = meta.regularMarketPrice || 0;

            const prev  = meta.chartPreviousClose || price;

            const change = prev > 0 ? ((price - prev) / prev) * 100 : 0;


            const info = { price, change, name: meta.shortName||meta.longName||tMap[tk]||tk, curr: meta.currency==='KRW'?'KRW':'USD' };

            cache[tk] = info;

            return info;

        } catch(e) {

            return { price:0, change:0, name:tMap[tk]||tk, curr:/^\d{6}$/.test(tk)?'KRW':'USD' };

        }

    }


    async function handleTk(u) {

        const tkIn = document.getElementById(`tk-${u}`);

        const nmIn = document.getElementById(`nm-${u}`);

        const prIn = document.getElementById(`pr-${u}`);

        const crSel = document.getElementById(`cr-${u}`);


        const tk = tkIn.value.trim().toUpperCase();

        if (!tk) { nmIn.value = ''; return; }


        tkIn.value = tk;

        nmIn.placeholder = "검색 중..."; nmIn.value = "";


        const info = await getStk(tk);


        nmIn.value = info.name;

        crSel.value = info.curr;

        if (info.price > 0 && !prIn.value) prIn.value = info.price;

        nmIn.placeholder = "자동 입력";

    }

    async function saveTr(u) {

        const eid = document.getElementById(`eid-${u}`).value;

        const tk  = document.getElementById(`tk-${u}`).value.toUpperCase();

        const data = {

            action:   eid ? 'edit' : 'add',

            id:       eid ? String(eid) : String(Date.now()),

            user:     u,

            date:     document.getElementById(`dt-${u}`).value,

            ticker:   tk,

            name:     document.getElementById(`nm-${u}`).value || tk,

            type:     document.getElementById(`ty-${u}`).value,

            currency: document.getElementById(`cr-${u}`).value,

            price:    parseFloat(document.getElementById(`pr-${u}`).value),

            qty:      parseFloat(document.getElementById(`qt-${u}`).value)

        };

        if (!data.ticker || isNaN(data.price) || isNaN(data.qty)) return alert("입력값을 확인해 주세요.");


        toggleLd(true, "스프레드시트에 기록 중...");

        try {

            await postReq(data);

            await fetchAll();

            cancelEd(u);

        } catch(e) {

            alert("저장 실패: " + e.message);

        }

        toggleLd(false);

    }


    function editTr(u, id) {

        const tr = state.trades[u].find(t => String(t.id) === String(id));

        if (!tr) return;

        document.getElementById(`eid-${u}`).value  = tr.id;

        document.getElementById(`dt-${u}`).value   = tr.date;

        document.getElementById(`tk-${u}`).value   = tr.ticker;

        document.getElementById(`nm-${u}`).value   = tr.name;

        document.getElementById(`ty-${u}`).value   = tr.type;

        document.getElementById(`cr-${u}`).value   = tr.currency;

        document.getElementById(`pr-${u}`).value   = tr.price;

        document.getElementById(`qt-${u}`).value   = tr.qty;

        document.getElementById(`btn-${u}`).innerText = "수정 완료";

        document.getElementById(`cbtn-${u}`).style.display = "block";

        window.scrollTo({ top: 0, behavior: 'smooth' });

    }


    function cancelEd(u) {

        document.getElementById(`eid-${u}`).value = "";

        ['tk','nm','pr','qt'].forEach(id => document.getElementById(`${id}-${u}`).value = '');

        document.getElementById(`dt-${u}`).value = new Date().toISOString().substring(0, 10);

        document.getElementById(`btn-${u}`).innerText = "저장";

        document.getElementById(`cbtn-${u}`).style.display = "none";

    }

    function updateAll() {

        let stats = [], hlds = [];


        for (const u of users) {

            let port = {}, real = 0, tInv = 0, tVal = 0, cLbl = [], cDat = [];


            state.trades[u].sort((a,b) => new Date(a.date)-new Date(b.date)).forEach(t => {

                if (!port[t.ticker]) port[t.ticker] = { n: t.name, q: 0, c: 0, cr: t.currency };

                let p = port[t.ticker];

                if (t.type === 'buy') {

                    p.c = (p.q*p.c + t.qty*t.price) / (p.q + t.qty);

                    p.q += t.qty;

                } else {

                    real += (t.price - p.c) * t.qty * (t.currency==='USD' ? exKRW : 1);

                    p.q  -= t.qty;

                }

            });


            let ptHtml = '';

            for (const [tk, p] of Object.entries(port)) {

                if (p.q <= 0) continue;

                const info  = cache[tk] || { price: p.c, name: p.n };

                const curPr = info.price > 0 ? info.price : p.c;

                const xr    = p.cr === 'USD' ? exKRW : 1;


                const iKw = p.q * p.c * xr;

                const vKw = p.q * curPr * xr;

                tInv += iKw; tVal += vKw;


                const ret  = iKw > 0 ? ((vKw - iKw)/iKw)*100 : 0;

                const sym  = p.cr === 'USD' ? '$' : '₩';

                const cls  = ret > 0 ? 'profit' : (ret < 0 ? 'loss' : '');

                const sign = ret > 0 ? '+' : '';

                const fn   = info.name || p.n;


                ptHtml += `<tr>

                    <td>${fn}<br><small style="color:#888">${tk}</small></td>

                    <td>${p.q.toLocaleString(undefined,{maximumFractionDigits:4})}</td>

                    <td>${sym}${p.c.toLocaleString(undefined,{maximumFractionDigits:2})}</td>

                    <td>${sym}${curPr.toLocaleString(undefined,{maximumFractionDigits:2})}</td>

                    <td>₩${vKw.toLocaleString('ko-KR',{maximumFractionDigits:0})}</td>

                    <td class="${cls}">${sign}${ret.toFixed(2)}%</td>

                </tr>`;

                cLbl.push(fn); cDat.push(vKw); hlds.push({ u, n: fn, r: ret });

            }


            document.getElementById(`port-${u}`).innerHTML =

                ptHtml || '<tr><td colspan="6" style="text-align:center;color:#aaa;padding:20px">종목 없음</td></tr>';


            // 매매 일지 렌더링

            const logArr = [...state.trades[u]].sort((a,b) => new Date(b.date)-new Date(a.date));

            document.getElementById(`log-${u}`).innerHTML = logArr.map(t => {

                const sym      = t.currency === 'USD' ? '$' : '₩';

                const typeCls  = t.type === 'buy' ? 'loss' : 'profit';

                const typeName = t.type === 'buy' ? '매수' : '매도';

                // data-* 속성으로 ID 전달 → onclick 인라인 문자열 이스케이프 문제 방지

                return `<tr>

                    <td>${t.date}</td>

                    <td>${t.name}</td>

                    <td class="${typeCls}">${typeName}</td>

                    <td>${sym}${t.price.toLocaleString(undefined,{maximumFractionDigits:2})}</td>

                    <td>${t.qty}</td>

                    <td>

                        <button class="a-btn edit-btn" data-u="${u}" data-id="${t.id}" onclick="editTr(this.dataset.u, this.dataset.id)">수정</button>

                        <button class="a-btn del-btn"  data-u="${u}" data-id="${t.id}" onclick="delTr(this.dataset.u, this.dataset.id)">삭제</button>

                    </td>

                </tr>`;

            }).join('') || '<tr><td colspan="6" style="text-align:center;color:#aaa;padding:20px">내역 없음</td></tr>';


            if (state.charts[u]) state.charts[u].destroy();

            state.charts[u] = new Chart(document.getElementById(`ch-${u}`).getContext('2d'), {

                type: 'doughnut',

                data: {

                    labels: cLbl,

                    datasets: [{

                        data: cDat,

                        backgroundColor: ['#4e79a7','#f28e2c','#e15759','#76b7b2','#59a14f','#edc949'],

                        borderWidth: 0

                    }]

                },

                options: {

                    maintainAspectRatio: false,

                    plugins: { legend: { position: 'right', labels: { boxWidth: 12 } } }

                }

            });


            const ur = tVal - tInv;

            const rp = tInv > 0 ? (ur/tInv)*100 : 0;

            stats.push({ u, i: tInv, v: tVal, ur, r: real, rp });

        }


        document.getElementById('dash-body').innerHTML = stats.map(s => {

            const urCls = s.ur > 0 ? 'profit' : (s.ur < 0 ? 'loss' : '');

            const rCls  = s.r  > 0 ? 'profit' : (s.r  < 0 ? 'loss' : '');

            const rpCls = s.rp > 0 ? 'profit' : (s.rp < 0 ? 'loss' : '');

            return `<tr>

                <td><strong>${s.u}</strong></td>

                <td>₩${s.i.toLocaleString('ko-KR',{maximumFractionDigits:0})}</td>

                <td>₩${s.v.toLocaleString('ko-KR',{maximumFractionDigits:0})}</td>

                <td class="${urCls}">₩${s.ur.toLocaleString('ko-KR',{maximumFractionDigits:0})}</td>

                <td class="${rCls}">₩${s.r.toLocaleString('ko-KR',{maximumFractionDigits:0})}</td>

                <td class="${rpCls}">${s.rp>0?'+':''}${s.rp.toFixed(2)}%</td>

            </tr>`;

        }).join('');


        const win = [...stats].sort((a,b) => b.rp - a.rp)[0];

        document.getElementById('dash-top').innerText =

            win && win.i > 0 ? `${win.u} (${win.rp>0?'+':''}${win.rp.toFixed(2)}%)` : '-';


        if (hlds.length > 0) {

            const b = hlds.reduce((p,c) => p.r > c.r ? p : c);

            const w = hlds.reduce((p,c) => p.r < c.r ? p : c);

            document.getElementById('dash-best').innerText  = `${b.n} (+${b.r.toFixed(2)}%)`;

            document.getElementById('dash-worst').innerText = `${w.n} (${w.r.toFixed(2)}%)`;

        } else {

            document.getElementById('dash-best').innerText  = '-';

            document.getElementById('dash-worst').innerText = '-';

        }

    }


    function switchTab(id, btn) {

        document.querySelectorAll('.tab-content, .tab-btn').forEach(el => el.classList.remove('active'));

        const target = id === 'dashboard' ? 'dashboard' : `u-${id.split('-')[1]}`;

        document.getElementById(target).classList.add('active');

        btn.classList.add('active');

    }
